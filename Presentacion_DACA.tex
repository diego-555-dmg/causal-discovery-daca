  \begin{columns}[T]

    \item \textbf{Descubrimiento causal:} inferir el grafo dirigido acíclico (DAG)

    \item Reformulado como \textbf{optimización continua}:
          \begin{itemize}\footnotesize
            \item NOTEARS (lineal) y \hl{DAG-GNN} (red neuronal en grafo).

% ---------- 3. Varsortability ----------
\begin{frame}{La trampa: \emph{varsortability} (Reisach et al., 2021)}
  \begin{itemize}
    \item En los grafos sintéticos habituales, la \textbf{varianza marginal crece}
          a lo largo del orden causal.
    \item Ordenar por varianza ya ``filtra'' el orden causal: un método trivial
          (\hl{sortnregress}) iguala a los sofisticados.
    \item \alert{Sospecha:} el desempeño no mide razonamiento causal, sino
          sensibilidad a la \textbf{escala}.
  \end{itemize}
  \vfill
  \begin{key}
  \centering
  \textbf{Varsortability} $v\in[0,1]$:\quad
  $v=\num{1}\Rightarrow$ la varianza revela el orden causal \quad$\bullet$\quad
  $v=\num{0.5}\Rightarrow$ azar
  \end{key}
\end{frame}

% ---------- 4. Pregunta ----------
\begin{frame}{Pregunta de investigación y contribución}
  \begin{block}{Pregunta}
  ¿\emph{Bajo qué condiciones de los datos} la flexibilidad de DAG-GNN produce
  mejoras \textbf{reales} sobre baselines no neuronales?
  \end{block}
  \textbf{Contribución --- un protocolo experimental controlado que:}
  \begin{enumerate}
    \item \textbf{varía la varsortability} vía la política de escalas de ruido;
    \item \textbf{separa} \emph{esqueleto} de \emph{orientación};
    \item contrasta datos \textbf{crudos vs.\ estandarizados};
    \item se \textbf{valida en datos reales} (red de proteínas de Sachs, 2005).
  \end{enumerate}
  \centering\footnotesize\textcolor{ink!55}{Implementación reproducible que mejora el código de Reisach et al.}
\end{frame}

% ---------- 5. Notación I ----------
\begin{frame}{Notación I: modelo causal, DAG y lo identificable}
  \begin{itemize}
    \item \textbf{Modelo Causal Estructural (MCE):} $\langle U,V,F,P(U)\rangle$;
          ecuaciones $X_j=f_j(\mathrm{PA}_j,U_j)\Rightarrow$ un \textbf{DAG}.
    \item \textbf{Límite de identificabilidad:} con datos observacionales
          (Markov $+$ fidelidad) solo se identifica la \textbf{clase de equivalencia
          de Markov} $=$ un \hl{CPDAG}.
  \end{itemize}
  \vfill
  \begin{key}
  \small Dos DAGs con el \textbf{mismo esqueleto} y las mismas \textbf{v-estructuras}
  ($a\!\to\!c\!\leftarrow\!b$) son \emph{indistinguibles} de los datos.
  $\Rightarrow$ separamos \hl{esqueleto} (¿hay arista?) de \hl{orientación} (¿en qué sentido?).
  \end{key}
\end{frame}

% ---------- 6. Notación II ----------
\begin{frame}{Notación II: datos y definición de \emph{varsortability}}
  \begin{columns}[T]
  \begin{column}{0.48\textwidth}
  \textbf{SEM lineal (ANM):}
  \[ X = X\,W + E \]
  \footnotesize $W$: pesos $i\!\to\!j$;\; $E$: ruido independiente.
  \normalsize\\[6pt]
  \textbf{Política de escalas del ruido} $\Rightarrow$ controla $v$:
  \begin{itemize}\footnotesize
    \item \emph{increasing}: $\mathrm{Var}$ crece con la profundidad $\to v\!\to\!1$
    \item \emph{equal} / \emph{random}
  \end{itemize}
  \end{column}
  \begin{column}{0.52\textwidth}
  \textbf{Varsortability:}
  \[ v=\frac{\displaystyle\sum_{k=1}^{d-1}\sum_{i\to j\in E^{k}}
      \mathbf{1}\!\left[\tfrac{\mathrm{Var}(X_j)}{\mathrm{Var}(X_i)}>1\right]}
      {\displaystyle\sum_{k=1}^{d-1}\sum_{i\to j\in E^{k}} 1} \]
  \footnotesize (empates $=\tfrac12$). Acuerdo entre el orden de varianzas y el
  orden causal a lo largo de caminos $E^k$.
  \end{column}
  \end{columns}
  \vspace{2pt}
  \begin{warn}
  \centering\small \textbf{Estandarizar} (media 0, var.\ 1) $\Rightarrow v=\num{0.5}$: se borra la señal de escala.
  \end{warn}
\end{frame}

% ---------- 7. Métodos ----------
\begin{frame}{Métodos comparados (1 principal $+$ 4 baselines)}
  \small
  \begin{tabular}{@{}>{\raggedright\arraybackslash}p{2.5cm} p{8.4cm}@{}}
  \toprule
  \hl{sortnregress} & baseline \emph{trivial}: ordena por varianza y regresa (Lasso-BIC). \\[2pt]
  \hl{NOTEARS} & opt.\ continua lineal con $h(W)=\mathrm{tr}(e^{W\circ W})-d=0$. \\[2pt]
  \hl{DAG-GNN} & VAE en grafo; $h(A)=\mathrm{tr}\!\big[(I+\tfrac1d A\!\circ\!A)^d\big]-d=0$. \\[2pt]
  \hl{PC / GES} & clásicos (restricciones / \emph{score}); \textbf{invariantes a escala} $\to$ CPDAG. \\
  \bottomrule
  \end{tabular}
  \vfill
  \begin{key}
  \footnotesize $h(\cdot)=0 \iff$ la matriz de pesos codifica un \textbf{DAG}.
  DAG-GNN lo optimiza con \textbf{Lagrangiano aumentado}:
  $\min\ \mathrm{ELBO} + \alpha\,h + \tfrac{\rho}{2}h^2$. Las MLP por nodo aportan la
  ``flexibilidad'' cuyo beneficio \emph{real} evaluamos.
  \end{key}
\end{frame}

% ---------- 8. Métricas ----------
\begin{frame}{Métricas: separar esqueleto de orientación}
  \begin{columns}[T]
  \begin{column}{0.62\textwidth}
  \begin{itemize}
    \item \hl{SHD}: aristas faltantes $+$ sobrantes $+$ invertidas.
          \textcolor{ink!55}{Menor es mejor.}
    \item \hl{F1 de esqueleto:} $F_1=\dfrac{2PR}{P+R}$ sobre aristas
          \emph{no dirigidas}.
    \item \hl{Precisión de orientación:} fracción bien orientada entre las
          detectadas (no dirigida del CPDAG $=0.5$).
  \end{itemize}
  \end{column}
  \begin{column}{0.38\textwidth}
  \begin{warn}
  \footnotesize \textbf{Por qué separar:} un método puede acertar el \emph{esqueleto}
  y aún así \emph{orientar al azar}. El SHD solo no lo revela.
  \end{warn}
  \end{column}
  \end{columns}
\end{frame}

% ---------- 14. Conclusiones (standout) ----------
\begin{frame}{Conclusiones}
  \begin{enumerate}
    \item En \emph{benchmarks} ANM lineales, la superioridad de NOTEARS/DAG-GNN es
          \hl{instrumental}: depende de una alta varsortability que un baseline
          trivial también explota.
    \item Al \textbf{estandarizar}, la ventaja se evapora; \hl{PC y GES} resultan preferibles.
    \item En \textbf{datos reales} (Sachs, $v=\num{0.59}$) la ventaja no aparece.
  \end{enumerate}
  \vfill
  \begin{block}{Recomendaciones metodológicas}
  \footnotesize
  (i) estandarizar como \textbf{control estándar}; (ii) reportar \textbf{esqueleto y
  orientación por separado}; (iii) la evidencia \emph{no} respalda adoptar DAG-GNN
  por su flexibilidad no lineal en este régimen.
  \end{block}
\end{frame}
